generator client {

datasource db {

  url      = env("DATABASE_URL")

model Account {
  id                String  @id @default(cuid())
  userId            String
  type              String
  provider          String
  providerAccountId String
  refresh_token     String?
  access_token      String?
  expires_at        Int?
  token_type        String?
  scope             String?
  id_token          String?
  session_state     String?
  user              User    @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([provider, providerAccountId])
  @@map("accounts")
}

model Session {
  id           String   @id @default(cuid())
  sessionToken String   @unique
  userId       String
  expires      DateTime
  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@map("sessions")
}

model User {
  id            String      @id @default(cuid())
  name          String?
  email         String?     @unique
  emailVerified DateTime?
  image         String?
  role          Role        @default(USER)
  igrejaId      String?
  senha         String
  createdAt     DateTime    @default(now())
  updatedAt     DateTime    @updatedAt
  cargoId       String?
  accounts      Account[]
  igrejasAdmin  Igreja?     @relation("IgrejaAdmin")
  permissoes    Permissao[]
  sessions      Session[]
  cargo         Cargo?      @relation(fields: [cargoId], references: [id])
  igreja        Igreja?     @relation(fields: [igrejaId], references: [id])

  @@map("users")
}

model Igreja {
  id                  String                 @id @default(cuid())
  nome                String
  slug                String                 @unique
  plano               Plano                  @default(FREE)
  adminId             String?                @unique
  membros             Int                    @default(1)
  createdAt           DateTime               @default(now())
  updatedAt           DateTime               @updatedAt
  cargos              Cargo[]
  eventos             Evento[]
  admin               User?                  @relation("IgrejaAdmin", fields: [adminId], references: [id])
  publico             IgrejaPublico?
  cronogramaAnual     CronogramaAnual[]
  membrosLista        Membro[]
  users               User[]
  ebdTurmas           EscolaDominicalTurma[]
  backupConfig        BackupConfig?
  backupGlobalTargets BackupGlobalConfig[]   @relation("BackupGlobalTargetIgreja")

  @@map("igrejas")
}

model Cargo {
  id        String   @id @default(cuid())
  nome      String
  igrejaId  String
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  igreja    Igreja   @relation(fields: [igrejaId], references: [id])
  users     User[]

  @@unique([igrejaId, nome])
  @@map("cargos")
}

model Permissao {
  id           String   @id @default(cuid())
  userId       String
  recurso      String
  ler          Boolean  @default(false)
  criar        Boolean  @default(false)
  editar       Boolean  @default(false)
  deletar      Boolean  @default(false)
  compartilhar Boolean  @default(false)
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, recurso])
  @@map("permissoes")
}

model Membro {
  id                     String    @id @default(cuid())
  igrejaId               String
  numeroSequencial       Int
  nome                   String
  telefone               String?
  endereco               String?
  numeroEndereco         String?
  bairro                 String?
  cidade                 String?
  estado                 String?
  estadoCivil            String?
  nomeMae                String?
  nomePai                String?
  rg                     String?
  cpf                    String?   @db.VarChar(11)
  cargo                  String
  dataNascimento         DateTime?
  dataBatismo            DateTime?
  numeroCarteirinha      String?
  dataCriacaoCarteirinha DateTime?
  dataVencCarteirinha    DateTime?
  createdAt              DateTime  @default(now())
  updatedAt              DateTime  @updatedAt
  observacoes            String?

  ativo Boolean @default(true)

  igreja Igreja @relation(fields: [igrejaId], references: [id], onDelete: Cascade)

  ebdTurmasComoProfessor EscolaDominicalTurma[]      @relation("ProfessorEbd")
  ebdTurmasAluno         EscolaDominicalTurmaAluno[]
  ebdFrequencias         EscolaDominicalFrequencia[]

  @@unique([igrejaId, numeroSequencial])
  @@index([igrejaId])
  @@index([igrejaId, nome])
  @@index([igrejaId, cargo])
  @@index([igrejaId, dataVencCarteirinha])
  @@map("membros")
}

model BackupConfig {
  id             String    @id @default(cuid())

  hora           String    @default("02:00")

  @@map("backup_configs")
}

}

model EscolaDominicalTurma {
  id           String   @id @default(cuid())
  igrejaId     String
  nome         String
  departamento String?
  professorId  String
  ativa        Boolean  @default(true)
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt

  igreja    Igreja                          @relation(fields: [igrejaId], references: [id], onDelete: Cascade)
  professor Membro                          @relation("ProfessorEbd", fields: [professorId], references: [id])
  alunos    EscolaDominicalTurmaAluno[]
  registros EscolaDominicalRegistroMensal[]

  @@unique([igrejaId, nome])
  @@index([igrejaId])
  @@index([professorId])
  @@map("ebd_turmas")
}

model EscolaDominicalTurmaAluno {
  id        String   @id @default(cuid())
  turmaId   String
  membroId  String
  createdAt DateTime @default(now())

  turma  EscolaDominicalTurma @relation(fields: [turmaId], references: [id], onDelete: Cascade)
  membro Membro               @relation(fields: [membroId], references: [id], onDelete: Cascade)

  @@unique([turmaId, membroId])
  @@index([membroId])
  @@map("ebd_turma_alunos")
}

model EscolaDominicalRegistroMensal {
  id          String   @id @default(cuid())
  turmaId     String
  mes         Int
  ano         Int
  observacoes String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  turma       EscolaDominicalTurma        @relation(fields: [turmaId], references: [id], onDelete: Cascade)
  frequencias EscolaDominicalFrequencia[]
  domingos    EscolaDominicalDomingo[]

  @@unique([turmaId, mes, ano])
  @@index([mes, ano])
  @@map("ebd_registros_mensais")
}

model EscolaDominicalDomingo {
  id               String   @id @default(cuid())
  registroMensalId String
  domingoNumero    Int
  visitantes       Int      @default(0)
  oferta           Decimal? @db.Decimal(10, 2)
  revistasLivros   Int      @default(0)
  observacao       String?
  isFolgaGeral     Boolean @default(false)
  motivoFolga      String?
  createdAt        DateTime @default(now())
  updatedAt        DateTime @updatedAt

  registroMensal EscolaDominicalRegistroMensal @relation(fields: [registroMensalId], references: [id], onDelete: Cascade)

  @@unique([registroMensalId, domingoNumero])
  @@map("ebd_domingos")
}

model EscolaDominicalFrequencia {
  id               String    @id @default(cuid())
  registroMensalId String
  membroId         String
  domingoNumero    Int
  status           EbdStatus @default(FALTA)
  createdAt        DateTime  @default(now())
  updatedAt        DateTime  @updatedAt

  registroMensal EscolaDominicalRegistroMensal @relation(fields: [registroMensalId], references: [id], onDelete: Cascade)
  membro         Membro                        @relation(fields: [membroId], references: [id], onDelete: Cascade)

  @@unique([registroMensalId, membroId, domingoNumero])
  @@index([membroId])
  @@index([domingoNumero])
  @@map("ebd_frequencias")
}
